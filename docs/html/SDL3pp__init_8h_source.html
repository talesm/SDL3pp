1117inline bool SDL::updateActive(bool active)
1118{
1119 static std::atomic_bool currentlyInitd{false};
1120 bool result = !currentlyInitd.exchange(active);
1121 if (active && !result) {
1122 SetErrorUnformatted(
1123 "Can not initialize, there is already an active instance");
1124 } else
1125 m_active = active;
1126 return result;
1127}
