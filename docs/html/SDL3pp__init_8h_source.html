907inline bool SDL::updateActive(bool active)
908{
909 static std::atomic_bool currentlyInitd{false};
910 bool result = !currentlyInitd.exchange(active);
911 if (active && !result) {
912 SetErrorUnformatted(
913 "Can not initialize, there is already an active instance");
914 } else
915 m_active = active;
916 return result;
917}
