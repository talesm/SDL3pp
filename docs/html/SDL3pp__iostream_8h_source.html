374 std::string Read(size_t size = -1)
375 {
376 Sint64 pos = Tell();
377 auto curSize = SDL_GetIOSize(get());
378 if ((curSize < 0 || pos < 0)) {
379 if (size == size_t(-1)) return {};
380 } else if (curSize - pos <= 0) {
381 return {};
382 } else if (curSize - pos < size) {
383 size = curSize - pos;
384 }
385 std::string result(size, 0);
386 auto actualSize = Read(result);
387 if (actualSize < size) result.resize(actualSize);
388 return result;
389 }
