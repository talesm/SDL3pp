229 std::string Read(size_t size = -1)
230 {
231 Sint64 pos = Tell();
232 auto curSize = SDL_GetIOSize(get());
233 if ((curSize < 0 || pos < 0)) {
234 if (size == size_t(-1)) return {};
235 } else if (curSize - pos <= 0) {
236 return {};
237 } else if (curSize - pos < size) {
238 size = curSize - pos;
239 }
240 std::string result(size, 0);
241 auto actualSize = Read(result);
242 if (actualSize < size) result.resize(actualSize);
243 return result;
244 }
