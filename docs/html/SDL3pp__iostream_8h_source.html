496 std::string Read(size_t size = -1)
497 {
498 std::string result;
499 Sint64 pos = Tell();
500 auto curSize = SDL_GetIOSize(get());
501 if ((curSize < 0 || pos < 0)) {
502 if (size == size_t(-1)) return {};
503 } else if (curSize - pos <= 0) {
504 return {};
505 } else if (curSize - pos < size) {
506 size = curSize - pos;
507 }
508 result.resize(size);
509 auto actualSize = Read(result.data(), size);
510 if (actualSize < size) result.resize(actualSize);
511 return result;
512 }
