232 std::string Read(size_t size = -1)
233 {
234 Sint64 pos = Tell();
235 auto curSize = SDL_GetIOSize(get());
236 if ((curSize < 0 || pos < 0)) {
237 if (size == size_t(-1)) return {};
238 } else if (curSize - pos <= 0) {
239 return {};
240 } else if (curSize - pos < size) {
241 size = curSize - pos;
242 }
243 std::string result(size, 0);
244 auto actualSize = Read(result);
245 if (actualSize < size) result.resize(actualSize);
246 return result;
247 }
