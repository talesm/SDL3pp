561 std::string Read(size_t size = -1)
562 {
563 Sint64 pos = Tell();
564 auto curSize = SDL_GetIOSize(get());
565 if ((curSize < 0 || pos < 0)) {
566 if (size == size_t(-1)) return {};
567 } else if (curSize - pos <= 0) {
568 return {};
569 } else if (curSize - pos < size) {
570 size = curSize - pos;
571 }
572 std::string result(size, 0);
573 auto actualSize = Read(result);
574 if (actualSize < size) result.resize(actualSize);
575 return result;
576 }
