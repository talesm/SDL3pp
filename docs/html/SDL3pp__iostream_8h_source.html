228 std::string Read(size_t size = -1)
229 {
230 Sint64 pos = Tell();
231 auto curSize = SDL_GetIOSize(get());
232 if ((curSize < 0 || pos < 0)) {
233 if (size == size_t(-1)) return {};
234 } else if (curSize - pos <= 0) {
235 return {};
236 } else if (curSize - pos < size) {
237 size = curSize - pos;
238 }
239 std::string result(size, 0);
240 auto actualSize = Read(result);
241 if (actualSize < size) result.resize(actualSize);
242 return result;
243 }
