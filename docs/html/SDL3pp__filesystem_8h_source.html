71 Path& operator/=(std::string_view other)
72 {
73 if (empty()) {
74 reset(Path(other).release());
75 } else if (!other.empty()) {
76 size_t lhsSz = size();
77 size_t rhsSz = other.size();
78 size_t finalSize = lhsSz + lhsSz + 1;
79 bool trailingDir = back() != '/' && back() != '\\';
80 if (trailingDir) { finalSize += 1; }
81 auto newBuf = static_cast<char*>(malloc(finalSize));
82 memcpy(newBuf, data(), lhsSz);
83 newBuf += lhsSz;
84 if (trailingDir) *(newBuf++) = '/';
85 memcpy(newBuf, other.data(), rhsSz);
86 newBuf += rhsSz;
87 *newBuf = 0;
88 reset(newBuf, finalSize - 1);
89 }
90 return *this;
91 }
