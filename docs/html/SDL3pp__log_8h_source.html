1048inline LogOutputCB GetLogOutputFunction()
1049{
1050 using Wrapper = UniqueCallbackWrapper<LogOutputCB>;
1051 LogOutputFunction cb;
1052 void* userdata;
1053 GetLogOutputFunction(&cb, &userdata);
1054 if (userdata == nullptr) {
1055 return [cb](LogCategory c, LogPriority p, StringParam m) {
1056 cb(nullptr, c, p, m);
1057 };
1058 }
1059 if (auto cb = Wrapper::at(userdata)) return cb;
1060 return [cb, userdata](LogCategory c, LogPriority p, StringParam m) {
1061 cb(userdata, c, p, m);
