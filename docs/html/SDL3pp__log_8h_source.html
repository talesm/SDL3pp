1049inline LogOutputCB GetLogOutputFunction()
1050{
1051 using Wrapper = UniqueCallbackWrapper<LogOutputCB>;
1052 LogOutputFunction cb;
1053 void* userdata;
1054 GetLogOutputFunction(&cb, &userdata);
1055 if (userdata == nullptr) {
1056 return [cb](LogCategory c, LogPriority p, StringParam m) {
1057 cb(nullptr, c, p, m);
1058 };
1059 }
1060 if (auto cb = Wrapper::at(userdata)) return cb;
1061 return [cb, userdata](LogCategory c, LogPriority p, StringParam m) {
1062 cb(userdata, c, p, m);
1063 };
1064}
