704inline LogOutputFunctionCB GetLogOutputFunction()
705{
706 using Wrapper = UniqueWrapper<LogOutputFunctionCB>;
707 LogOutputFunction cb;
708 void* userdata;
709 if (userdata == nullptr) {
710 return [cb](LogCategory c, LogPriority p, StringParam m) {
711 cb(nullptr, c, p, m);
712 };
713 }
714 GetLogOutputFunction(&cb, &userdata);
715 if (!Wrapper::contains(userdata)) {
716 return [cb, userdata](LogCategory c, LogPriority p, StringParam m) {
717 cb(userdata, c, p, m);
718 };
719 }
720 return Wrapper::at(userdata);
721}
