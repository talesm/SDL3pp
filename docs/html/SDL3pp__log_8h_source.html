715inline LogOutputFunctionCB GetLogOutputFunction()
716{
717 using Wrapper = UniqueWrapper<LogOutputFunctionCB>;
718 LogOutputFunction cb;
719 void* userdata;
720 if (userdata == nullptr) {
721 return [cb](LogCategory c, LogPriority p, StringParam m) {
722 cb(nullptr, c, p, m);
723 };
724 }
725 GetLogOutputFunction(&cb, &userdata);
726 if (!Wrapper::contains(userdata)) {
727 return [cb, userdata](LogCategory c, LogPriority p, StringParam m) {
728 cb(userdata, c, p, m);
729 };
730 }
731 return Wrapper::at(userdata);
732}
