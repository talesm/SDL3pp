355inline TimerID AddTimer(std::chrono::nanoseconds interval, TimerCB callback)
356{
357 using Wrapper = CallbackWrapper<TimerCB>;
358 using Store = KeyValueWrapper<TimerID, TimerCB*>;
359
360 auto cb = Wrapper::Wrap(std::move(callback));
361
362 if (TimerID id = SDL_AddTimerNS(
363 interval.count(),
364 [](void* userdata, TimerID timerID, Uint64 interval) -> Uint64 {
365 auto& f = *static_cast<TimerCB*>(userdata);
366 auto next = f(timerID, std::chrono::nanoseconds(interval)).count();
367 // If ask to removal, then remove it
368 if (next == 0) delete Store::release(timerID);
369 return next;
370 },
371 cb)) {
372 Store::Wrap(id, std::move(cb));
373 return id;
374 }
375 delete cb;
376 return TimerID{0};
377}
