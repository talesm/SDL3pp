323{
324 using Wrapper = CallbackWrapper<TimerCB>;
325 using Store = KeyValueWrapper<TimerID, TimerCB*>;
326
327 auto cb = Wrapper::Wrap(std::move(callback));
328
329 if (TimerID id = SDL_AddTimerNS(
330 interval.count(),

334 // If ask to removal, then remove it
